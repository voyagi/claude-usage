  <action>
Create `src/commands/exportData.ts` with a function `exportUsageData` that:

1. Takes `store: UsageStore` and `planType: string` as parameters
2. Loads data from store via `store.loadUsageData()`
3. If no data, shows info message "No usage data to export" and returns
4. Builds export object with dual format:

```typescript
import * as vscode from 'vscode';
import type { UsageStore } from '../storage/usageStore.js';
import { serializeTimeBuckets } from '../aggregation/timeBuckets.js';
import { formatTokens } from '../ui/formatting.js';
import { getClaudeProjectsDir } from '../utils/paths.js';

export async function exportUsageData(store: UsageStore, planType: string): Promise<void> {
  const loaded = await store.loadUsageData();
  if (!loaded) {
    vscode.window.showInformationMessage('Claude Usage: No usage data to export.');
    return;
  }

  const { buckets, stats } = loaded;

  // Build summary (human-friendly)
  let totalInput = 0, totalOutput = 0, totalCacheCreation = 0, totalCacheRead = 0, totalCost = 0;
  for (const agg of buckets.daily.values()) {
    totalInput += agg.inputTokens;
    totalOutput += agg.outputTokens;
    totalCacheCreation += agg.cacheCreationTokens;
    totalCacheRead += agg.cacheReadTokens;
    totalCost += agg.totalCost;
  }

  const exportData = {
    exportedAt: new Date().toISOString(),
    extensionVersion: '0.1.0',
    dataSource: getClaudeProjectsDir(),
    planType,
    summary: {
      totalInputTokens: totalInput,
      totalOutputTokens: totalOutput,
      totalCacheCreationTokens: totalCacheCreation,
      totalCacheReadTokens: totalCacheRead,
      totalCost: Math.round(totalCost * 100) / 100,
      filesProcessed: stats.filesProcessed,
      linesSkipped: stats.linesSkipped,
      sessionCount: buckets.session.size,
      dailyBucketCount: buckets.daily.size,
    },
    raw: serializeTimeBuckets(buckets),
  };

  // Show save dialog
  const uri = await vscode.window.showSaveDialog({
    defaultUri: vscode.Uri.file(`claude-usage-export-${new Date().toISOString().slice(0, 10)}.json`),
    filters: { 'JSON': ['json'] },
  });

  if (!uri) return; // User cancelled

  const content = new TextEncoder().encode(JSON.stringify(exportData, null, 2));
  await vscode.workspace.fs.writeFile(uri, content);
  vscode.window.setStatusBarMessage(`Claude Usage: Exported to ${uri.fsPath}`, 5000);
}
```

Use `workspace.fs.writeFile` (not Node.js fs) for remote/SSH compatibility as recommended in RESEARCH.md. Use `TextEncoder` for encoding.

The `serializeTimeBuckets` function already exists in `src/aggregation/timeBuckets.ts` and converts Maps to arrays -- reuse it for the raw section.

Import note: `formatTokens` from `src/ui/formatting.ts` is available but may not be needed in the export (JSON is machine-readable). Keep the export numeric, not formatted strings.
