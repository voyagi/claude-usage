    ```typescript
    import * as fs from 'fs/promises';
    import * as path from 'path';
    import * as os from 'os';

    import type { PlanType } from '../types.js';
    ```

    **Class: `CredentialsWatcher`**

    Constructor takes `context: vscode.ExtensionContext` and `onTierChange: (tier: PlanType) => void`.

    Methods:
    - `start(fallbackTier: PlanType): Promise<PlanType>` — reads credentials.json once on startup, sets up FileSystemWatcher, returns detected tier (or fallback). Uses `vscode.workspace.createFileSystemWatcher(new vscode.RelativePattern(credDir, '.credentials.json'))`. Watches for `onDidChange` and `onDidCreate`. Push watcher to `context.subscriptions`.
    - `private async readAndDetectTier(fallback: PlanType): Promise<PlanType>` — reads file at `path.join(os.homedir(), '.claude', '.credentials.json')`, calls `parseCredentialsFile`, then `detectTierFromCredentials`. Wrapped in try/catch — returns fallback on any error. Logs via Logger.
    - `dispose(): void` — cleanup (watcher auto-disposed via context.subscriptions).

    **Important:** The FileSystemWatcher `onDidChange` handler should call `readAndDetectTier` and then call `onTierChange` only if the detected tier differs from the last known tier (avoid redundant callbacks). Store `lastKnownTier: PlanType` as instance field.

    **Edge cases:**
    - File doesn't exist: return fallback, no error logged (normal for fresh installs)
    - File is empty or malformed JSON: log warning, return fallback
    - File exists but has no rateLimitTier/subscriptionType: return fallback

    **Part A: rateLimitDetector.ts**

    Create a module that parses rate limit error events from JSONL lines and provides logic to refine limit estimates.

    ```typescript
    export interface RateLimitEvent {
      timestamp: Date;
      limitType: 'session' | 'weekly' | 'unknown';
      errorMessage: string;
    }
    ```

    `parseRateLimitEvent(line: string): RateLimitEvent | null`
    - Parse JSON from line. If it contains `type: "error"` and `error.type: "rate_limit_error"`, extract timestamp and classify limit type.
    - Classification from error.message (case-insensitive):
      - Contains "daily" or "weekly" -> 'weekly'
      - Contains "per-minute" or "rpm" or "session" -> 'session'
      - Otherwise -> 'unknown'
    - Return null if not a rate limit error or parse fails.
    - Use `z.object` from Zod for safe parsing (consistent with project conventions).

    `refineLimitEstimate(currentEstimate: number, observedUsage: number): number`
    - When a 429 is observed, the true limit is at or below `observedUsage`.
    - Compute `observedLimit = Math.floor(observedUsage * 0.95)` (5% safety margin).
    - Return `Math.min(currentEstimate, observedLimit)` — only adjust downward.
    - If observedUsage <= 0, return currentEstimate unchanged.

    No VS Code dependencies. Pure data module.

    **Part B: package.json configuration**
