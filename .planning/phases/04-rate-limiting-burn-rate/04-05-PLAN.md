    Create a rate limit event handler:
    ```typescript
    function handleRateLimitEvent(event: RateLimitEvent): void {
      const plan = getPlanConfig(getSelectedPlan());
      const current = refinedLimits ?? {};

      if (event.limitType === 'session') {
        const currentLimit = current.sessionTokenLimit ?? plan.sessionTokenLimit ?? 0;
        // Get current session output tokens from the last known buckets
        // Use the session tokens that were active when the 429 hit
        const refined = refineLimitEstimate(currentLimit, currentLimit); // We refine based on observed usage
        if (refined < currentLimit) {
          current.sessionTokenLimit = refined;
          logger.info(`Refined session limit: ${currentLimit} -> ${refined}`);
        }
      } else if (event.limitType === 'weekly') {
        const currentLimit = current.weeklyTokenLimit ?? plan.weeklyTokenLimit ?? 0;
        const refined = refineLimitEstimate(currentLimit, currentLimit);
        if (refined < currentLimit) {
          current.weeklyTokenLimit = refined;
          logger.info(`Refined weekly limit: ${currentLimit} -> ${refined}`);
        }
      }

      // Update timestamp and persist
      refinedLimits = { ...current, lastUpdated: new Date().toISOString() } as RefinedLimits;
      saveRefinedLimits(refinedLimits).catch((err) => {
        logger.error(`Failed to save refined limits: ${err.message}`);
      });

      // Trigger refresh to apply new limits
      vscode.commands.executeCommand('claude-usage.refresh');
    }
    ```

    **IMPORTANT CORRECTION:** The above handler needs the ACTUAL token usage at the time
    of the 429 event, not the current limit. The right approach:

    Track current session/weekly output tokens in a module-level variable that gets
    updated on every onUpdate callback. Then when a 429 hits, use that observed usage
    as the `observedUsage` parameter to `refineLimitEstimate`.

    Add module-level tracking:

    let lastKnownSessionTokens = 0;
    let lastKnownWeeklyTokens = 0;

    In the SessionWatcher onUpdate callback (where buildStatusBarData is called), capture:

    lastKnownSessionTokens = rateLimitStatus.session5h.currentTokens;
    lastKnownWeeklyTokens = rateLimitStatus.weekly.currentTokens;

    Fix the handler to use observed usage:
    ```typescript
    function handleRateLimitEvent(event: RateLimitEvent): void {
      const plan = getPlanConfig(getSelectedPlan());

      if (event.limitType === 'session' && lastKnownSessionTokens > 0) {
        const currentLimit = refinedLimits?.sessionTokenLimit ?? plan.sessionTokenLimit ?? 0;
        const refined = refineLimitEstimate(currentLimit, lastKnownSessionTokens);
        if (refined < currentLimit) {
          refinedLimits = {
            ...refinedLimits,
            sessionTokenLimit: refined,
            lastUpdated: new Date().toISOString(),
          } as RefinedLimits;
          logger.info(`Refined session limit: ${currentLimit} -> ${refined} (observed: ${lastKnownSessionTokens})`);
        }
      } else if (event.limitType === 'weekly' && lastKnownWeeklyTokens > 0) {
        const currentLimit = refinedLimits?.weeklyTokenLimit ?? plan.weeklyTokenLimit ?? 0;
        const refined = refineLimitEstimate(currentLimit, lastKnownWeeklyTokens);
        if (refined < currentLimit) {
          refinedLimits = {
            ...refinedLimits,
            weeklyTokenLimit: refined,
            lastUpdated: new Date().toISOString(),
          } as RefinedLimits;
          logger.info(`Refined weekly limit: ${currentLimit} -> ${refined} (observed: ${lastKnownWeeklyTokens})`);
        }
      }

      // Persist and refresh
      if (refinedLimits) {
        saveRefinedLimits(refinedLimits).catch((err) => {
          logger.error(`Failed to save refined limits: ${err.message}`);
        });
        vscode.commands.executeCommand('claude-usage.refresh');
      }
    }
    ```
